APIs that developers love

get-transactions.js
import RivoFi from 'nexuslink-node';

const client = new RivoFi({
clientId: process.env.NEXUS_CLIENT_ID,
secret: process.env.NEXUS_SECRET,
env: 'production'
});

// Fetch enriched transactions
const { transactions } = await client.transactions.get({
access_token: "access_tok_x9f...",
start_date: "2026-04-01",
end_date: "2026-05-01"
});

console.log(`Found ${transactions.length} transactions`);
get_transactions.py
from nexuslink import Client

client = Client(
client_id=os.environ["NEXUS_CLIENT_ID"],
secret=os.environ["NEXUS_SECRET"],
environment="production"
)

# Fetch enriched transactions
response = client.transactions.get(
access_token="access_tok_x9f...",
start_date="2026-04-01",
end_date="2026-05-01"
)

print(f"Found {len(response.transactions)} transactions")
